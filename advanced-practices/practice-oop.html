Задача 2: Класс RectangleЛёгкая

Создайте класс Rectangle с атрибутами width и height. Добавьте методы area() (площадь) и perimeter() (периметр).

Ввод:width=5, height=3
Вывод:Площадь: 15, Периметр: 16
class Rectangle:
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

Задача 3: Класс BankAccountСредняя

Создайте класс BankAccount с приватным атрибутом __balance. Добавьте методы deposit(amount), withdraw(amount) и get_balance().

💡 Подсказка

Используйте два подчёркивания для приватного атрибута. Проверяйте достаточно ли средств при снятии.

Вывод:Баланс: 1500 После пополнения: 2000 После снятия: 1700
class BankAccount:
    def __init__(self, initial_balance=0):
        self.__balance = initial_balance
    
    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount
            return f"Внесено {amount}"
        return "Некорректная сумма"
    
    def withdraw(self, amount):
        if 0 < amount <= self.__balance:
            self.__balance -= amount
            return f"Снято {amount}"
        return "Недостаточно средств"
    
    def get_balance(self):
        return self.__balance

acc = BankAccount(1500)
print(f"Баланс: {acc.get_balance()}")
acc.deposit(500)
print(f"После пополнения: {acc.get_balance()}")
acc.withdraw(300)
print(f"После снятия: {acc.get_balance()}")

Задача 6: ФигурыСредняя

Создайте базовый класс Shape с методом area(). Создайте дочерние классы Circle, Square, Triangle со своими реализациями площади.

💡 Подсказка

Импортируйте math.pi для круга. Для треугольника используйте формулу Герона.

Вывод:Круг (r=5): 78.54 Квадрат (a=4): 16 Треугольник (3,4,5): 6.0
import math

class Shape:
    def area(self):
        return 0

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        return math.pi * self.radius ** 2

class Square(Shape):
    def __init__(self, side):
        self.side = side
    
    def area(self):
        return self.side ** 2

class Triangle(Shape):
    def __init__(self, a, b, c):
        self.a, self.b, self.c = a, b, c
    
    def area(self):
        s = (self.a + self.b + self.c) / 2
        return math.sqrt(s * (s-self.a) * (s-self.b) * (s-self.c))

circle = Circle(5)
square = Square(4)
triangle = Triangle(3, 4, 5)
print(f"Круг (r=5): {circle.area():.2f}")
print(f"Квадрат (a=4): {square.area()}")
print(f"Треугольник (3,4,5): {triangle.area()}")

Задача 7: Магические методыСредняя

Создайте класс Book с магическими методами __str__, __repr__, __len__, __eq__.

Вывод:'Война и мир' by Толстой Book('Война и мир', 'Толстой', 1200) Страниц: 1200
class Book:
    def __init__(self, title, author, pages):
        self.title = title
        self.author = author
        self.pages = pages
    
    def __str__(self):
        return f"'{self.title}' by {self.author}"
    
    def __repr__(self):
        return f"Book('{self.title}', '{self.author}', {self.pages})"
    
    def __len__(self):
        return self.pages
    
    def __eq__(self, other):
        return self.title == other.title and self.author == other.author

book = Book("Война и мир", "Толстой", 1200)
print(book)
print(repr(book))
print(f"Страниц: {len(book)}")

Создайте класс Student с атрибутами name, grades (список оценок). Добавьте методы для добавления оценки, получения среднего балла и определения статуса (отличник/хорошист/троечник).

Вывод:Средний балл: 4.4 Статус: Хорошист
class Student:
    def __init__(self, name):
        self.name = name
        self.grades = []
    
    def add_grade(self, grade):
        if 1 <= grade <= 5:
            self.grades.append(grade)
    
    def get_average(self):
        if not self.grades:
            return 0
        return sum(self.grades) / len(self.grades)
    
    def get_status(self):
        avg = self.get_average()
        if avg >= 4.5:
            return "Отличник"
        elif avg >= 3.5:
            return "Хорошист"
        elif avg >= 2.5:
            return "Троечник"
        else:
            return "Отстающий"

student.add_grade(4)

student.add_grade(5)

Создайте класс Library, который хранит список книг. Реализуйте методы: add_book(), remove_book(), find_by_author(), get_available_books().

class Book:
    def __init__(self, title, author, year):
        self.title = title
        self.author = author
        self.year = year
        self.is_available = True
    
    def __str__(self):
        status = "доступна" if self.is_available else "выдана"
        return f"'{self.title}' ({self.year}) - {status}"

class Library:
    def __init__(self, name):
        self.name = name
        self.books = []
    
    def add_book(self, book):
        self.books.append(book)
        print(f"Книга '{book.title}' добавлена")
    
    def remove_book(self, title):
        for book in self.books:
            if book.title == title:
                self.books.remove(book)
                print(f"Книга '{title}' удалена")
                return
        print("Книга не найдена")
    
    def find_by_author(self, author):
        return [b for b in self.books if b.author == author]
    
    def get_available_books(self):
        return [b for b in self.books if b.is_available]

lib = Library("Городская")
lib.add_book(Book("Война и мир", "Толстой", 1869))
lib.add_book(Book("Анна Каренина", "Толстой", 1877))
lib.add_book(Book("Преступление и наказание", "Достоевский", 1866))

print(f"\nКниги Толстого:")
for book in lib.find_by_author("Толстой"):
    print(f"  - {book}")
Задача 11: Интернет-магазинСложная

Создайте систему классов для интернет-магазина: Product, Cart, Customer. Реализуйте добавление товаров в корзину, подсчёт общей стоимости.

class Product:
    def __init__(self, name, price, quantity=1):
        self.name = name
        self.price = price
        self.quantity = quantity
    
    def get_total(self):
        return self.price * self.quantity

class Cart:
    def __init__(self):
        self.items = []
    
    def add_product(self, product):
        self.items.append(product)
    
    def get_total(self):
        return sum(item.get_total() for item in self.items)
    
    def show_cart(self):
        if not self.items:
            print("Корзина пуста")
            return
        print("Ваша корзина:")
        for item in self.items:
            print(f"  {item.name} x{item.quantity} = {item.get_total()} руб.")
        print(f"Итого: {self.get_total()} руб.")

class Customer:
    def __init__(self, name):
        self.name = name
        self.cart = Cart()

customer = Customer("Иван")
customer.cart.add_product(Product("Ноутбук", 50000))
customer.cart.add_product(Product("Мышь", 1500, 2))
customer.cart.add_product(Product("Клавиатура", 3500))
customer.cart.show_cart()
Задача 12: УниверситетСложная

Создайте систему управления университетом с классами Person, Student, Professor, Course. Реализуйте enrolment на курсы и ведение оценок.

class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

class Student(Person):
    def __init__(self, name, age, student_id):
        super().__init__(name, age)
        self.student_id = student_id
        self.courses = []
        self.grades = {}
    
    def enroll(self, course):
        self.courses.append(course)
        course.students.append(self)
        self.grades[course.name] = []
    
    def add_grade(self, course_name, grade):
        if course_name in self.grades:
            self.grades[course_name].append(grade)
    
    def get_gpa(self, course_name):
        grades = self.grades.get(course_name, [])
        return sum(grades) / len(grades) if grades else 0

class Professor(Person):
    def __init__(self, name, age, title):
        super().__init__(name, age)
        self.title = title
        self.courses = []

class Course:
    def __init__(self, name, code):
        self.name = name
        self.code = code
        self.students = []
        self.professor = None
    
    def assign_professor(self, professor):
        self.professor = professor
        professor.courses.append(self)

# Использование
prof = Professor("Петр Иванов", 45, "Доцент")
course = Course("Программирование", "CS101")
course.assign_professor(prof)

student = Student("Алексей", 20, "S001")
student.enroll(course)
student.add_grade("Программирование", 5)
student.add_grade("Программирование", 4)
student.add_grade("Программирование", 5)

print(f"Студент: {student.name}")
print(f"Курс: {course.name}")
print(f"GPA: {student.get_gpa('Программирование'):.2f}")
