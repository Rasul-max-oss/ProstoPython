Задача 7: Поиск по авторуСредняя

Создайте функцию find_by_author(author), которая возвращает все книги указанного автора.

Ввод:Толстой
Вывод:Найдено книг: 2 - Война и мир (1865) - Анна Каренина (1877)
import sqlite3

def find_by_author(author):
    conn = sqlite3.connect('library.db')
    cursor = conn.cursor()
    
    cursor.execute('''
        SELECT title, year FROM books 
        WHERE author LIKE ?
    ''', (f'%{author}%',))
    
    books = cursor.fetchall()
    conn.close()
    
    print(f"Найдено книг: {len(books)}")
    for title, year in books:
        print(f"- {title} ({year})")
    
    return books

find_by_author("Толстой")

Задача 8: Статистика по авторамСредняя

Выведите количество книг и среднее количество страниц для каждого автора.

Вывод:Толстой: 2 книг(и), ср. 1032 стр. Достоевский: 1 книг(и), ср. 672 стр. ...
import sqlite3

conn = sqlite3.connect('library.db')
cursor = conn.cursor()

cursor.execute('''
    SELECT author, 
           COUNT(*) as book_count,
           AVG(pages) as avg_pages
    FROM books
    GROUP BY author
    ORDER BY book_count DESC
''')

for author, count, avg_pages in cursor.fetchall():
    print(f"{author}: {count} книг(и), ср. {avg_pages:.0f} стр.")

conn.close()

Задача 11: JOIN запросСложная

Выведите информацию о выданных книгах: имя читателя, название книги, дата выдачи.

Вывод:Иван Петров взял "Война и мир" (2024-01-15)
import sqlite3

conn = sqlite3.connect('library.db')
cursor = conn.cursor()

cursor.execute('''
    SELECT r.name, b.title, l.loan_date
    FROM loans l
    JOIN readers r ON l.reader_id = r.id
    JOIN books b ON l.book_id = b.id
''')

for reader, book, date in cursor.fetchall():
    print(f"{reader} взял \"{book}\" ({date})")

conn.close()
Задача 12: Система учётаСложная

Создайте класс LibraryDB с методами: add_book(), add_reader(), issue_book(), return_book(), get_available_books().

import sqlite3
from datetime import datetime

class LibraryDB:
    def __init__(self, db_name='library.db'):
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.create_tables()
    
    def create_tables(self):
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS books (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                title TEXT NOT NULL,
                author TEXT,
                is_available INTEGER DEFAULT 1
            )
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS readers (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL
            )
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS loans (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                reader_id INTEGER,
                book_id INTEGER,
                loan_date TEXT,
                return_date TEXT,
                FOREIGN KEY (reader_id) REFERENCES readers(id),
                FOREIGN KEY (book_id) REFERENCES books(id)
            )
        ''')
        self.conn.commit()
    
    def add_book(self, title, author):
        self.cursor.execute('''
            INSERT INTO books (title, author) VALUES (?, ?)
        ''', (title, author))
        self.conn.commit()
        print(f"Книга '{title}' добавлена")
    
    def add_reader(self, name):
        self.cursor.execute('''
            INSERT INTO readers (name) VALUES (?)
        ''', (name,))
        self.conn.commit()
        print(f"Читатель '{name}' добавлен")
    
    def issue_book(self, reader_id, book_id):
        self.cursor.execute('SELECT is_available FROM books WHERE id = ?', (book_id,))
        result = self.cursor.fetchone()
        if result and result[0] == 1:
            self.cursor.execute('''
                INSERT INTO loans (reader_id, book_id, loan_date)
                VALUES (?, ?, ?)
            ''', (reader_id, book_id, datetime.now().strftime('%Y-%m-%d')))
            self.cursor.execute('''
                UPDATE books SET is_available = 0 WHERE id = ?
            ''', (book_id,))
            self.conn.commit()
            print(f"Книга #{book_id} выдана читателю #{reader_id}")
        else:
            print("Книга недоступна")
    
    def return_book(self, book_id):
        self.cursor.execute('''
            UPDATE loans SET return_date = ? 
            WHERE book_id = ? AND return_date IS NULL
        ''', (datetime.now().strftime('%Y-%m-%d'), book_id))
        self.cursor.execute('''
            UPDATE books SET is_available = 1 WHERE id = ?
        ''', (book_id,))
        self.conn.commit()
        print(f"Книга #{book_id} возвращена")
    
    def get_available_books(self):
        self.cursor.execute('SELECT id, title, author FROM books WHERE is_available = 1')
        return self.cursor.fetchall()
    
    def close(self):
        self.conn.close()

# Использование
lib = LibraryDB(':memory:')  # База в памяти
lib.add_book("Война и мир", "Толстой")
lib.add_book("Преступление и наказание", "Достоевский")
lib.add_reader("Иван Петров")
lib.issue_book(1, 1)
print("Доступные книги:", lib.get_available_books())
lib.return_book(1)
lib.close()
