SQLite — Базы данных

База данных — это организованное хранилище информации. SQLite — лёгкая встраиваемая база данных, которая хранится в одном файле и не требует сервера.

Что такое база данных

Представьте таблицу Excel, но с возможностью быстро искать, сортировать и связывать данные между несколькими таблицами.

Основные понятия

Таблица — хранит данные (как лист Excel)
Строка (запись) — одна запись в таблице
Столбец (поле) — атрибут данных (имя, возраст, цена)
Первичный ключ (PRIMARY KEY) — уникальное поле для идентификации строки
SQL — язык запросов к базе данных

Типы данных в SQLite

Справка Типы данных 
INTEGER     — целые числа (1, 42, -100)
REAL        — числа с плавающей точкой (3.14, -0.5)
TEXT        — текст (строки)
BLOB        — бинарные данные (изображения, файлы)
BOOLEAN     — логические значения (0 или 1)

Подключение к базе данных

В Python есть встроенный модуль sqlite3 для работы с SQLite.

main.py Пример 1: Подключение 
import sqlite3

# Подключаемся к базе (файл создастся, если его нет)
conn = sqlite3.connect('my_database.db')

# Создаём курсор — объект для выполнения запросов
cursor = conn.cursor()

print("База данных подключена!")

# Закрываем соединение
conn.close()

Вывод программы:

Терминал Результат 
База данных подключена!
.db — стандартное расширение для файлов SQLite.
Всегда закрывайте соединение через conn.close() или используйте with.

Создание таблицы

Перед добавлением данных нужно создать таблицу с нужными полями.

main.py Пример 2: Создание таблицы 
import sqlite3

# Используем контекстный менеджер — соединение закроется автоматически
with sqlite3.connect('company.db') as conn:
    cursor = conn.cursor()
    
    # Создаём таблицу employees
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS employees (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            age INTEGER,
            city TEXT,
            salary REAL
        )
    ''')
    
    # Сохраняем изменения
    conn.commit()
    
print("Таблица 'employees' создана!")

Вывод программы:

Терминал Результат 
Таблица 'employees' создана!

Разбор SQL-запроса

CREATE TABLE IF NOT EXISTS — создать таблицу, если она не существует
id INTEGER PRIMARY KEY AUTOINCREMENT — поле id, целое число, первичный ключ, увеличивается автоматически
name TEXT NOT NULL — поле name, текст, обязательное (не может быть пустым)
age INTEGER — поле age, целое число
salary REAL — поле salary, число с плавающей точкой

Добавление данных (INSERT)

Вставляем записи в таблицу.

main.py Пример 3: Вставка данных 
import sqlite3

with sqlite3.connect('company.db') as conn:
    cursor = conn.cursor()
    
    # Вставка одной записи
    cursor.execute('''
        INSERT INTO employees (name, age, city, salary)
        VALUES (?, ?, ?, ?)
    ''', ('Иван Петров', 28, 'Москва', 75000))
    
    # Вставка нескольких записей
    employees = [
        ('Анна Смирнова', 32, 'СПб', 85000),
        ('Борис Козлов', 25, 'Москва', 65000),
        ('Виктория Лебедева', 35, 'Казань', 95000),
        ('Дмитрий Волков', 42, 'СПб', 120000)
    ]
    
    cursor.executemany('''
        INSERT INTO employees (name, age, city, salary)
        VALUES (?, ?, ?, ?)
    ''', employees)
    
    conn.commit()
    
print("Добавлено 5 сотрудников")

Вывод программы:

Терминал Результат 
Добавлено 5 сотрудников
Знаки вопроса ? — это параметры для защиты от SQL-инъекций. Никогда не подставляйте значения напрямую в SQL-запрос.

import sqlite3

with sqlite3.connect('company.db') as conn:
    cursor = conn.cursor()
    
    # Выбор всех записей
    cursor.execute('SELECT * FROM employees')
    rows = cursor.fetchall()  # Получить все строки
    
    print("Все сотрудники:")
    for row in rows:
        print(f"  {row[0]}. {row[1]} ({row[2]} лет), {row[3]}, {row[4]} руб.")
    
    # Выбор с условием
    print("\nСотрудники из Москвы:")
    cursor.execute("SELECT name, salary FROM employees WHERE city = 'Москва'")
    moscow = cursor.fetchall()
    for name, salary in moscow:
        print(f"  {name}: {salary} руб.")
    
    # Выбор с сортировкой
    print("\nСотрудники по убыванию зарплаты:")
    cursor.execute("SELECT name, salary FROM employees ORDER BY salary DESC")
    sorted_employees = cursor.fetchall()
    for name, salary in sorted_employees:
        print(f"  {name}: {salary} руб.")

Агрегатные функции

SQL позволяет выполнять вычисления над данными.

main.py Пример 7: Агрегатные функции 
import sqlite3

with sqlite3.connect('company.db') as conn:
    cursor = conn.cursor()
    
    # Количество сотрудников
    cursor.execute('SELECT COUNT(*) FROM employees')
    print(f"Всего сотрудников: {cursor.fetchone()[0]}")
    
    # Средняя зарплата
    cursor.execute('SELECT AVG(salary) FROM employees')
    print(f"Средняя зарплата: {cursor.fetchone()[0]:.0f} руб.")
    
    # Максимальная зарплата
    cursor.execute('SELECT MAX(salary) FROM employees')
    print(f"Максимальная зарплата: {cursor.fetchone()[0]} руб.")
    
    # Минимальная зарплата
    cursor.execute('SELECT MIN(salary) FROM employees')
    print(f"Минимальная зарплата: {cursor.fetchone()[0]} руб.")
    
    # Общая сумма зарплат
    cursor.execute('SELECT SUM(salary) FROM employees')
    print(f"Общий фонд зарплат: {cursor.fetchone()[0]:,.0f} руб.")
    
    # Группировка по городу
    print("\nСотрудников по городам:")
    cursor.execute('''
        SELECT city, COUNT(*) as count, AVG(salary) as avg_salary
        FROM employees 
        GROUP BY city
    ''')
    for city, count, avg in cursor.fetchall():
        print(f"  {city}: {count} чел., средняя {avg:.0f} руб.")

Связанные таблицы

В реальных проектах данные хранятся в нескольких связанных таблицах.

main.py Пример 8: Две таблицы 
import sqlite3

with sqlite3.connect('company.db') as conn:
    cursor = conn.cursor()
    
    # Создаём таблицу должностей
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS positions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            min_salary REAL,
            max_salary REAL
        )
    ''')
    
    # Добавляем должности
    positions = [
        ('Менеджер', 50000, 80000),
        ('Разработчик', 80000, 150000),
        ('Дизайнер', 60000, 100000),
        ('Аналитик', 70000, 120000)
    ]
    cursor.executemany('INSERT INTO positions (title, min_salary, max_salary) VALUES (?, ?, ?)', positions)
    
    # Добавляем поле position_id в таблицу сотрудников
    try:
        cursor.execute('ALTER TABLE employees ADD COLUMN position_id INTEGER')
    except:
        pass  # Поле уже существует
    
    # Назначаем должности
    cursor.execute("UPDATE employees SET position_id = 2 WHERE name IN ('Иван Петров', 'Виктория Лебедева')")
    cursor.execute("UPDATE employees SET position_id = 1 WHERE name = 'Анна Смирнова'")
    cursor.execute("UPDATE employees SET position_id = 4 WHERE name = 'Дмитрий Волков'")
    
    conn.commit()
    
    # Запрос с объединением таблиц (JOIN)
    print("Сотрудники с должностями:")
    cursor.execute('''
        SELECT e.name, e.city, p.title, e.salary
        FROM employees e
        JOIN positions p ON e.position_id = p.id
    ''')
    for row in cursor.fetchall():
        print(f"  {row[0]} ({row[1]}) — {row[2]}, {row[3]} руб.")

Типы соединений (JOIN)

INNER JOIN — только совпадающие записи в обеих таблицах
LEFT JOIN — все записи из левой таблицы + совпадения из правой
RIGHT JOIN — все записи из правой таблицы + совпадения из левой
FULL JOIN — все записи из обеих таблиц

Практический проект: Система учёта задач

Создадим полноценное приложение для управления задачами (To-Do List).

todo.py Пример 9: Менеджер задач 
import sqlite3
from datetime import datetime

class TaskManager:
    def __init__(self, db_name='tasks.db'):
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.create_table()
    
    def create_table(self):
        """Создать таблицу задач"""
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS tasks (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                title TEXT NOT NULL,
                description TEXT,
                priority INTEGER DEFAULT 1,
                status TEXT DEFAULT 'new',
                created_at TEXT
            )
        ''')
        self.conn.commit()
    
    def add_task(self, title, description='', priority=1):
        """Добавить задачу"""
        created_at = datetime.now().strftime('%Y-%m-%d %H:%M')
        self.cursor.execute('''
            INSERT INTO tasks (title, description, priority, created_at)
            VALUES (?, ?, ?, ?)
        ''', (title, description, priority, created_at))
        self.conn.commit()
        print(f"Задача '{title}' добавлена!")
    
    def show_tasks(self, status=None):
        """Показать задачи"""
        if status:
            self.cursor.execute('''
                SELECT id, title, priority, status, created_at 
                FROM tasks 
                WHERE status = ?
                ORDER BY priority DESC, created_at
            ''', (status,))
        else:
            self.cursor.execute('''
                SELECT id, title, priority, status, created_at 
                FROM tasks 
                ORDER BY priority DESC, created_at
            ''')
        
        tasks = self.cursor.fetchall()
        
        if not tasks:
            print("Задач нет")
            return
        
        print(f"\n{'ID':<4} {'Название':<25} {'Приоритет':<10} {'Статус':<10} {'Дата'}")
        print("-" * 70)
        for task in tasks:
            print(f"{task[0]:<4} {task[1][:25]:<25} {task[2]:<10} {task[3]:<10} {task[4]}")
    
    def complete_task(self, task_id):
        """Отметить задачу как выполненную"""
        self.cursor.execute('''
            UPDATE tasks SET status = 'done' WHERE id = ?
        ''', (task_id,))
        self.conn.commit()
        print(f"Задача #{task_id} выполнена!")
    
    def delete_task(self, task_id):
        """Удалить задачу"""
        self.cursor.execute('DELETE FROM tasks WHERE id = ?', (task_id,))
        self.conn.commit()
        print(f"Задача #{task_id} удалена!")
    
    def close(self):
        self.conn.close()

# Использование
manager = TaskManager()

# Добавляем задачи
manager.add_task("Изучить SQLite", "Прочитать урок и сделать практику", priority=3)
manager.add_task("Купить продукты", "Молоко, хлеб, яйца", priority=1)
manager.add_task("Закончить проект", "Сдать код до пятницы", priority=5)
manager.add_task("Позвонить маме", priority=2)

# Показываем все задачи
print("\n=== ВСЕ ЗАДАЧИ ===")
manager.show_tasks()

# Показываем только новые
print("\n=== НОВЫЕ ЗАДАЧИ ===")
manager.show_tasks(status='new')

# Выполняем задачу
manager.complete_task(2)

# Показываем выполненные
print("\n=== ВЫПОЛНЕННЫЕ ===")
manager.show_tasks(status='done')

# Удаляем задачу
manager.delete_task(1)

manager.close()

Работа с Pandas и SQLite

Pandas может напрямую читать и записывать данные в SQLite.

analysis.py Пример 10: Pandas + SQLite 
import sqlite3
import pandas as pd

# Подключаемся к базе
conn = sqlite3.connect('company.db')

# Читаем данные в DataFrame
df = pd.read_sql_query('''
    SELECT e.name, e.city, e.salary, p.title as position
    FROM employees e
    JOIN positions p ON e.position_id = p.id
''', conn)

print("Данные из базы в Pandas:")
print(df)

# Статистика
print("\nСтатистика:")
print(f"Средняя зарплата: {df['salary'].mean():.0f} руб.")
print(f"Максимальная зарплата: {df['salary'].max()} руб.")

# Группировка по должностям
print("\nСредняя зарплата по должностям:")
print(df.groupby('position')['salary'].mean())

# Сохраняем результат обратно в SQLite
df.to_sql('employees_report', conn, if_exists='replace', index=False)
print("\nОтчёт сохранён в таблицу 'employees_report'")

conn.close()
